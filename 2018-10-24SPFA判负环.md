/*
第一种是记录从1到x的最短路包含的边数
cnt[1] = 0,在更新边的时候cnt[v] = cnt[u] + 1
如果发现cnt[v]>=n的时候就有负环

第二种判断点的入队次数，如果tot[u]>=n时,就有负环 
*/
#include<cstdio>
#include<queue>
#include<cstring>
#define MAXM 14000
#define MAXN 5000
#define mset(a,b) (memset(a,b,sizeof(a)))
using namespace std;
inline int read(){
	char c = getchar();
	int f = 1;
	while(c < '0' || '9' < c){
		if(c == '-')
			f = -1;
		c = getchar();
	}
	int x = 0;
	while('0' <= c && c <= '9'){
		x = 10 * x + c - '0';
		c = getchar(); 
	}
	return f * x;
}
struct Edge{
	int next,to,w;
}e[MAXM];
int ct,head[MAXN];
inline void addE(int u,int v,int w){
	e[++ct].to = v;
	e[ct].w = w;
	e[ct].next = head[u];
	head[u] = ct;
}
bool inq[MAXN];
int cnt[MAXN],tot[MAXN],d[MAXN];
int T,n,m,a,b,c;
bool spfa(int s){
	queue<int> q;
	mset(inq,0);mset(tot,0);mset(cnt,0);mset(d,0x3f);
	q.push(s);
	d[s] = 0;cnt[s] = 0;tot[s] = 1;inq[s] = 1;
	while(!q.empty()){
		int u = q.front();q.pop();
		inq[u] = false;
		for(int i = head[u];i;i = e[i].next){
			int v = e[i].to;
			int w = e[i].w;
			if(d[v] > d[u] + w){
				d[v] = d[u] + w;
				cnt[v] = cnt[u] + 1;
				if(cnt[v] >= n)
					return false;
				if(!inq[v]){
					q.push(v);
					tot[v]++;
					if(tot[v] >= n)
						return false;
					inq[v] = true;
				}
			}
		}
	}
	return true;
}
int main(){
	//freopen("testdata.in","r",stdin);
	T = read();
	while(T--){
		mset(head,0);mset(e,0);
		ct = 0;
		n = read();
		m = read();
		for(int i = 1;i <= m;i++){
			a = read();
			b = read();
			c = read();
			addE(a,b,c);
			if(c >= 0)
				addE(b,a,c);
		}
		printf("%s\n",spfa(1) ? "N0" : "YE5");
	}
	return 0;
}
