#include<cstdio>
#define max(a,b) (a > b ? a : b)
#define lowbit(x) (x & -x)
int c[500000],n,x,a[100000];
int ask(int p){
	int ans = 0;
	for(;p > 0;p -= lowbit(p))
		ans = max(ans,c[p]);
	return ans;
}
void add(int p,int x){
	for(;p <= n;p += lowbit(p))
		c[p] = max(c[p],x);
}
int main(){
	freopen("testdata.in","r",stdin);
	int ans = 0;
	scanf("%d",&n);
	for(int i = 1;i <= n;i++){
		scanf("%d",&x);
		int maxx = ask(x - 1) + 1;
		printf("%d ",maxx);
		add(x,maxx);
		ans = max(ans,maxx);
	}
	printf("\n%d",ans);
	return 0;
}
