~~~cpp
#include<cstdio>
#include<cstring>
#include<algorithm>
#define MAXN 250
using namespace std;
typedef unsigned long long ull;
ull hash[MAXN][MAXN],w[8][MAXN];
int p,k,s,n,f[MAXN][MAXN],cnt[MAXN][MAXN];
char str[MAXN],word[8][MAXN];
bool vis[MAXN];
int dfs(int l,int r){
	memset(vis,0,sizeof(vis));
	int tot = 0;
	for(int i = l;i <= r;i++){
		for(int j = 1;j <= s;j++){
			int len = strlen(word[j]) - 1;
			if(i + len <= r && hash[i][i + len] == w[j][len + 1]){
				tot++;
				break;
			}
		}
	}
	return tot;
}

int main(){
	//freopen("1026.in","r",stdin);
	scanf("%d%d",&p,&k);
	for(int i = 0;i < p;i++){
		scanf("%s",str + 20 * i + 1);
		n += 20;
	}
	for(int i = 1;i <= n;i++){
		for(int j = i;j <= n;j++){
			hash[i][j] = hash[i][j - 1] * 131 + str[j] - 'a' + 1;
		}
	}
	scanf("%d",&s);
	for(int i = 1;i <= s;i++){
		scanf("%s",word[i]);
		for(unsigned int j = 1;j <= strlen(word[i]);j++){
			w[i][j] = 131 * w[i][j - 1] + word[i][j - 1] - 'a' + 1;
		}
	}
	for(int i = 1;i <= n;i++){
		for(int j = i;j <= n;j++){
			cnt[i][j] = dfs(i,j);
		}
	}
	for(int i = 1;i <= n;i++)
		f[i][0] = f[i][1] = cnt[1][i];
	for(int i = 1;i <= n;i++){
		for(int j = 1;j <= k;j++){
			for(int x = j;x < i;x++){
				f[i][j] = max(f[i][j],f[x][j - 1] + cnt[x + 1][i]);
			}
		}
	}
	printf("%d\n",f[n][k]);
}
~~~
