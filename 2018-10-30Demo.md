```cpp
#include<cstdio>
int k,w,len,s;
long long ans[500];
long long f[15000][1000];
int sum(long long x){
	int len = 1;
	ans[1] += x;
	while(ans[len] >= 10){
		long long p = ans[len];
		ans[len] %= 10;
		ans[++len] += p / 10;
	}
	//printf("%d\n",len);
	ans[0] = len;
}
void printf(){
	for(int i = ans[0];i > 0;i--)
		printf("%d",ans[i]);
}
int main(){
	//freopen("digital.in","r",stdin);
	//freopen("digital.ans","w",stdout);
	sum(121);
	sum(3421);
	sum(3421);
	sum(3421);
	printf();
	return 0;
	scanf("%d%d",&k,&w);
	int x = (1 << (w % k)) - 1;//最前位
	if(!x){
		len--;
		x = (1 << k) - 1;
	}
	len += w / k;//表示除了最前面一位的长度
	s = (1 << k) - 1;//后面的位最大数 
	//printf("%d %d %d\n",x,s,len);
	for(int i = 1;i <= s;i++){
		f[2][i] = s - i;
		sum(f[2][i]);
	}
	if(len == 1){
		printf("%d\n",ans);
		return 0;
	}
	if(len == 2){
		for(int i = 1;i <= x;i++){
			for(int j = i + 1;j <= s;j++){
				f[3][i] += f[2][j];
			}
			sum(f[3][i]);
		}
	}else{
		for(int i = 3;i <= len;i++){//最后一位要特判
			for(int j = 1;j < s;j++){
				for(k = j + 1;k <= s;k++){
					f[i][j] += f[i - 1][k];
				}
				sum(f[i][j]);
			}
		}
		for(int i = 1;i <= x;i++){
			for(int j = i + 1;j <= s;j++)
				f[len + 1][i] += f[len][j];
			sum(f[len + 1][i]);
		}
	}
	printf();
	return 0;
}
```

```cpp
#include<cstdio>
int n,ans[10000];
bool check(int x){
	int tmp = 0;
	int t = x;
	while(t){
		tmp += t % 10;
		t /= 10;
	}
	return n == (x + tmp);
}
int main(){
	scanf("%d",&n);
	int size = 0;
	for(int i = 1;i <= n;i++){
		if(check(i)){
			ans[++size] = i;
		}
	}
	printf("%d\n",size);
	for(int i = 1;i <= size;i++){
		printf("%d\n",ans[i]);
	}
	return 0;
}
```

```cpp
#include<cstdio>
#define max(a,b) (a > b ? a : b)
typedef long long ll;
ll n,ans[1000];
bool check(ll x){
	int tmp = 0;
	ll t = x;
	while(t){
		tmp += t % 10;
		t /= 10;
	}
	return n == (x + tmp);
}
int main(){
	scanf("%lld",&n);
	int size = 0;
	for(ll i = max(n - 1000,0);i <= n;i++){
		if(check(i)){
			ans[++size] = i;
		}
	}
	printf("%d\n",size);
	for(int i = 1;i <= size;i++)
		printf("%lld\n",ans[i]);
	return 0;
}
```
